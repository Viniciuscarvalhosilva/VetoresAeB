# VetoresAeB
Exibir o valor de c
##  ALGORITMO
###  Início
	int x=0 , c=0 , mult=0
	int A[] = new int [5]
	int B[] = new int [5]
	enquanto x<=4 faça
		digite valores do vetor A[x]
		grava valor em cada posição de A[x]
	fim enquanto
	enquanto x<=4 faça
		digite valores do vetor B[x]
		grava valor em cada posição de B[x]
	fim enquanto
		enquanto x<5 faça
		mult = A[x]*B[x]
		c = c + mult
		fim enquanto
	fim enquanto
	Exibe valor de c
fim
